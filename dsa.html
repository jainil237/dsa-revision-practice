Sliding Window

Where Used

Contiguous subarrays/substrings — finding max/min length, max sum, or a specific condition within a window of elements.

Hints to Identify

"contiguous""subarray""substring""longest/shortest""maximum/minimum""at most K"

Core Logic

Maintain a window [left, right] using two pointers. Expand right to include elements. When window becomes invalid, shrink left. Track the best answer during the process.

Template Code 

int left = 0, bestResult = 0;
for (int right = 0; right < n; right++) {
    // Add arr[right] to window state
    // (update hashmap, sum, count, etc.)
    while (windowIsInvalid(left, right)) {
        // Remove arr[left] from window state
        left++;
    }
    bestResult = Math.max(bestResult, right - left + 1);
}
return bestResult;

Solved Examples

Medium#3 — Longest Substring Without Repeating Characters
Approach

Use a map to store the last index of each character. When we see a duplicate, move left to max(left, lastSeen[char] + 1). This ensures the window always has unique characters.

Step-by-step
1
Initialize left=0, maxLen=0, int[128] map for ASCII chars
2
Loop right from 0 to n-1: get char c = s.charAt(right)
3
If c was seen, left = max(left, map[c]) to skip past duplicate
4
Update map[c] = right + 1 (store next position)
5
maxLen = max(maxLen, right - left + 1)
class Solution {
    public int lengthOfLongestSubstring(String s) {
        int[] map = new int[128];
        int left = 0, maxLen = 0;
        for (int right = 0; right < s.length(); right++) {
            char c = s.charAt(right);
            left = Math.max(left, map[c]);
            map[c] = right + 1;
            maxLen = Math.max(maxLen, right - left + 1);
        }
        return maxLen;
    }
}
var lengthOfLongestSubstring = function(s) {
    const map = new Map();
    let left = 0, maxLen = 0;
    for (let right = 0; right < s.length; right++) {
        const c = s[right];
        if (map.has(c)) left = Math.max(left, map.get(c));
        map.set(c, right + 1);
        maxLen = Math.max(maxLen, right - left + 1);
    }
    return maxLen;
};

Two Pointers

Where Used

Sorted arrays, pairs/triplets with sum, palindrome checks, removing duplicates, in-place operations.

Hints to Identify

"sorted array""two sum""palindrome""pairs""in-place""O(1) space"

Core Logic

Place one pointer at start, one at end (or both at start). Move them based on comparison. Sort first if unsorted. Eliminates O(n²) to O(n).

Solved Examples

Medium#11 — Container With Most Water

Two pointers at ends. Area = min(h[l], h[r]) × (r-l). Move the shorter pointer inward — moving taller can never increase area.

class Solution {
    public int maxArea(int[] height) {
        int left = 0, right = height.length - 1, maxWater = 0;
        while (left < right) {
            int h = Math.min(height[left], height[right]);
            maxWater = Math.max(maxWater, h * (right - left));
            if (height[left] < height[right]) left++;
            else right--;
        }
        return maxWater;
    }
}

    let left = 0, right = height.length - 1, maxWater = 0;

    }

};
Medium#15 — 3Sum

Sort array. Fix element i, use two pointers on rest to find pairs summing to -nums[i]. Skip duplicates.

class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        Arrays.sort(nums);
        List<List<Integer>> res = new ArrayList<>();
        for (int i = 0; i < nums.length - 2; i++) {
            if (i > 0 && nums[i] == nums[i-1]) continue;
            int l = i + 1, r = nums.length - 1;
            while (l < r) {
                int sum = nums[i] + nums[l] + nums[r];
                if (sum == 0) {
                    res.add(Arrays.asList(nums[i], nums[l], nums[r]));
                    while (l < r && nums[l] == nums[l+1]) l++;
                    while (l < r && nums[r] == nums[r-1]) r--;
                    l++; r--;
                } else if (sum < 0) l++;
                else r--;
            }
        }
        return res;
    }
}
var threeSum = function(nums) {
    nums.sort((a, b) => a - b);

                res.push([nums[i], nums[l], nums[r]]);
                while (l < r && nums[l] === nums[l+1]) l++;
                while (l < r && nums[r] === nums[r-1]) r--;
                l++; r--;
            } else if (sum < 0) l++; else r--;
        }
    }
    return res;
};

Fast & Slow Pointers

Where Used

Linked list cycle detection, finding middle element, detecting cycles in number sequences, finding cycle start.

Hints to Identify

"cycle""loop""linked list""middle""happy number"

Core Logic

Slow moves 1 step, fast moves 2. If they meet → cycle. If fast reaches null → no cycle. To find cycle start: reset one to head, move both 1 step until they meet.

Solved Examples

Easy#141 — Linked List Cycle

Classic Floyd's algorithm. If fast reaches null, no cycle. If slow == fast, cycle exists.

public class Solution {
    public boolean hasCycle(ListNode head) {
        ListNode slow = head, fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) return true;
        }
        return false;
    }
}
var hasCycle = function(head) {
    let slow = head, fast = head;
    while (fast && fast.next) {
        slow = slow.next;
        fast = fast.next.next;
        if (slow === fast) return true;
    }
    return false;
};
Easy#202 — Happy Number

Replace number with sum of squares of digits. Use fast/slow to detect cycle in the sequence. If it reaches 1 → happy.

class Solution {
    public boolean isHappy(int n) {
        int slow = n, fast = n;
        do {
            slow = sumSq(slow);
            fast = sumSq(sumSq(fast));
        } while (slow != fast);
        return slow == 1;
    }
    private int sumSq(int n) {
        int sum = 0;
        while (n > 0) { int d = n % 10; sum += d * d; n /= 10; }
        return sum;
    }
}
var isHappy = function(n) {
    const sumSq = (n) => {
        let sum = 0;
        while (n > 0) { const d = n % 10; sum += d * d; n = Math.floor(n / 10); }
        return sum;
    };
    let slow = n, fast = n;
    do { slow = sumSq(slow); fast = sumSq(sumSq(fast)); } while (slow !== fast);
    return slow === 1;
};

Merge Intervals

Where Used

Overlapping intervals, meeting rooms, calendar scheduling, range-based problems.

Hints to Identify

"intervals""overlapping""merge""meeting rooms""schedule"

Core Logic

Sort by start. Compare current with previous: if overlapping (curr.start ≤ prev.end), merge by updating prev.end = max(prev.end, curr.end). Otherwise add to result.

Solved Examples

Medium#56 — Merge Intervals

Sort by start. Iterate: if current overlaps last in result, merge. Otherwise add as new interval.

class Solution {
    public int[][] merge(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> Integer.compare(a[0], b[0]));
        List<int[]> res = new ArrayList<>();
        for (int[] iv : intervals) {
            if (res.isEmpty() || res.get(res.size()-1)[1] < iv[0])
                res.add(iv);
            else
                res.get(res.size()-1)[1] = Math.max(res.get(res.size()-1)[1], iv[1]);
        }
        return res.toArray(new int[res.size()][]);
    }
}
var merge = function(intervals) {
    intervals.sort((a, b) => a[0] - b[0]);
    const res = [];
    for (const iv of intervals) {
        if (!res.length || res[res.length-1][1] < iv[0]) res.push(iv);
        else res[res.length-1][1] = Math.max(res[res.length-1][1], iv[1]);
    }
    return res;
};

Cyclic Sort

Where Used

Arrays with numbers in range [1,n] or [0,n]. Missing numbers, duplicates, first missing positive — O(n) time, O(1) space.

Hints to Identify

"range 1 to n""missing number""duplicate""O(1) extra space"

Core Logic

Place each number at index (value-1). While nums[i] != nums[nums[i]-1], swap. After sorting, scan for mismatches.

Solved Examples

Easy#268 — Missing Number

XOR all indices and numbers. Pairs cancel. Remaining value is the missing number.

class Solution {
    public int missingNumber(int[] nums) {
        int xor = 0;
        for (int i = 0; i < nums.length; i++) xor ^= i ^ nums[i];
        return xor ^ nums.length;
    }
}
var missingNumber = function(nums) {
    let xor = 0;
    for (let i = 0; i < nums.length; i++) xor ^= i ^ nums[i];
    return xor ^ nums.length;
};

Mark index (num-1) as negative. If already negative → duplicate found.

In-place Linked List Reversal

Where Used

Reversing all or part of a linked list, k-group reversal, palindrome check.

Hints to Identify

"reverse""linked list""in-place""k-group""between positions"

Core Logic

Three pointers: prev, curr, next. Save next before changing curr.next. Point curr.next to prev. Move prev and curr forward. Key: always save nextNode first.

Solved Examples

Easy#206 — Reverse Linked List

Foundation pattern. prev=null, curr=head. Save next, reverse link, advance.

class Solution {
    public ListNode reverseList(ListNode head) {
        ListNode prev = null, curr = head;
        while (curr != null) {
            ListNode next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        return prev;
    }
}

    let prev = null, curr = head;
    while (curr) {
        const next = curr.next;
        curr.next = prev;
        prev = curr;
        curr = next;
    }
    return prev;
};

Tree BFS (Level Order)

Where Used

Level-order traversal, max/min per level, zigzag traversal, shortest path in unweighted trees.

Hints to Identify

"level order""breadth-first""level by level""shortest path""zigzag"

Core Logic

Queue. Add root. While queue not empty: get level size, process that many nodes, enqueue children. Gives level-by-level access.

Solved Examples

Medium#102 — Binary Tree Level Order Traversal

Standard BFS. For each level, collect all node values into a list.

class Solution {
    public List<List<Integer>> levelOrder(TreeNode root) {
        List<List<Integer>> res = new ArrayList<>();
        if (root == null) return res;
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        while (!q.isEmpty()) {
            int size = q.size();
            List<Integer> level = new ArrayList<>();
            for (int i = 0; i < size; i++) {
                TreeNode node = q.poll();
                level.add(node.val);
                if (node.left != null) q.offer(node.left);
                if (node.right != null) q.offer(node.right);
            }
            res.add(level);
        }
        return res;
    }
}
var levelOrder = function(root) {
    if (!root) return [];
    const res = [], q = [root];
    while (q.length) {
        const size = q.length, level = [];
        for (let i = 0; i < size; i++) {
            const node = q.shift();
            level.push(node.val);
            if (node.left) q.push(node.left);
            if (node.right) q.push(node.right);
        }
        res.push(level);
    }
    return res;
};

Tree DFS

Where Used

Pre/In/Post-order traversal, path sum, max path sum, diameter, validating BST — bottom-up or top-down processing.

Hints to Identify

"path sum""diameter""height""pre/in/post order""BST property"

Core Logic

Recursive: process node, recurse left, recurse right. Return values bubble up (post-order). Parameters push info down (pre-order with path). Base: if node==null return 0/null.

Solved Examples

Easy#104 — Maximum Depth of Binary Tree

Depth = 1 + max(leftDepth, rightDepth). Simplest DFS pattern.

class Solution {
    public int maxDepth(TreeNode root) {
        if (root == null) return 0;
        return 1 + Math.max(maxDepth(root.left), maxDepth(root.right));
    }
}
var maxDepth = function(root) {
    if (!root) return 0;
    return 1 + Math.max(maxDepth(root.left), maxDepth(root.right));
};
Hard#124 — Binary Tree Maximum Path Sum

For each node, compute max gain downward. Path through node = val + leftGain + rightGain. Track global max. Return only one branch upward.

class Solution {
    private int maxSum = Integer.MIN_VALUE;
    public int maxPathSum(TreeNode root) {
        gain(root);
        return maxSum;

    private int gain(TreeNode node) {
        if (node == null) return 0;

        int right = Math.max(gain(node.right), 0);
        maxSum = Math.max(maxSum, node.val + left + right);
        return node.val + Math.max(left, right);
    }
}

Pattern 09

Backtracking

Where Used

Permutations, combinations, subsets, N-Queens, Sudoku, word search — all possible arrangements/selections.

Hints to Identify

"all possible""permutations""combinations""subsets""arrangements"

Core Logic

Choose → Explore → Undo. Make a choice, recurse, then backtrack (undo). Use "used" array/set to avoid revisiting. Add to result at right base case.

Solved Examples

Medium#46 — Permutations

Swap-based: for each position, try swapping with every element after it. Recurse. Swap back (backtrack). When index reaches end, save copy.

class Solution {
    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        backtrack(nums, 0, res);
        return res;
    }
    private void backtrack(int[] nums, int idx, List<List<Integer>> res) {
        if (idx == nums.length) {
            List<Integer> perm = new ArrayList<>();
            for (int n : nums) perm.add(n);
            res.add(perm); return;
        }
        for (int i = idx; i < nums.length; i++) {
            swap(nums, idx, i);
            backtrack(nums, idx + 1, res);
            swap(nums, idx, i);
        }
    }
    private void swap(int[] n, int i, int j) { int t=n[i]; n[i]=n[j]; n[j]=t; }
}
var permute = function(nums) {
    const res = [];
    const bt = (idx) => {
        if (idx === nums.length) { res.push([...nums]); return; }
        for (let i = idx; i < nums.length; i++) {
            [nums[idx], nums[i]] = [nums[i], nums[idx]];
            bt(idx + 1);
            [nums[idx], nums[i]] = [nums[i], nums[idx]];
        }
    };
    bt(0); return res;
};
Medium#78 — Subsets

At each index, make a choice: include nums[i] or don't. When index reaches end, save a copy of current path.

class Solution {
    public List<List<Integer>> subsets(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        backtrack(nums, 0, new ArrayList<>(), res);
        return res;
    }
    private void backtrack(int[] nums, int idx, List<Integer> path, List<List<Integer>> res) {
        res.add(new ArrayList<>(path));
        for (int i = idx; i < nums.length; i++) {
            path.add(nums[i]);
            backtrack(nums, i + 1, path, res);
            path.remove(path.size() - 1);
        }
    }
}

};

        if (target == 0) { res.add(new ArrayList<>(path)); return; }
        if (target < 0) return;
        for (int i = idx; i < c.length; i++) {
            path.add(c[i]);
            backtrack(c, target - c[i], i, path, res);
            path.remove(path.size() - 1);
        }
    }
}

        if (remain < 0) return;

Where Used

"graph""nodes and edges""shortest path""island""connected""topological sort""prerequisites""grid with 0/1"

Core Logic

Build adjacency list from edges. BFS for shortest path (unweighted) & level-by-level exploration. DFS for connectivity, cycle detection & deep exploration. Always track visited set to avoid infinite loops. For topological sort: use in-degree array with BFS (Kahn's algorithm).

Deep Dive: Understanding Graph Patterns

Step 1 вҖ” How to represent a graph?

An adjacency list is a map/array where each node points to its neighbors. This is the most common and efficient representation.

Step 2 вҖ” BFS Template (Shortest Path / Level Order)

BFS uses a Queue. It explores level by level. This guarantees the shortest path in unweighted graphs because it visits nodes in order of their distance from source.

// BFS from source node
boolean[] visited = new boolean[n];
Queue<Integer> queue = new LinkedList<>();
queue.offer(source);
visited[source] = true;
int distance = 0;

while (!queue.isEmpty()) {
    int size = queue.size();  // all nodes at current distance
    for (int i = 0; i < size; i++) {
        int node = queue.poll();
        // PROCESS node here (check if target, record, etc.)
        for (int neighbor : adj.get(node)) {
            if (!visited[neighbor]) {
                visited[neighbor] = true;
                queue.offer(neighbor);
            }
        }
    }
    distance++;  // increment after processing each level
}
Step 3 вҖ” DFS Template (Connectivity / Exploration)

DFS goes deep first, then backtracks. Use recursion (call stack) or explicit stack. Ideal for detecting cycles, counting components, and flood-fill problems.

// DFS from a node
boolean[] visited = new boolean[n];
void dfs(int node) {
    visited[node] = true;
    // PROCESS node here
    for (int neighbor : adj.get(node)) {
        if (!visited[neighbor]) {
            dfs(neighbor);  // go deep first
        }
    }
}
// For counting connected components:
int components = 0;
for (int i = 0; i < n; i++) {
    if (!visited[i]) { components++; dfs(i); }
}
Step 4 вҖ” Topological Sort (Directed Acyclic Graph)

Only works on DAGs. Track in-degree (how many edges point TO each node). Start with nodes having in-degree 0. Process them, reduce neighbors' in-degrees. When a neighbor's in-degree hits 0, add to queue. If all nodes processed вҶ’ no cycle.

// Kahn's Algorithm
int[] inDegree = new int[n];
for (int[] edge : edges) inDegree[edge[0]]++; // edge: [to, from]

Queue<Integer> q = new LinkedList<>();
for (int i = 0; i < n; i++) if (inDegree[i] == 0) q.offer(i);

int processed = 0;
while (!q.isEmpty()) {
    int node = q.poll();
    processed++;
    for (int neighbor : adj.get(node)) {
        if (--inDegree[neighbor] == 0) q.offer(neighbor);
    }
}
// if processed != n вҶ’ there's a cycle вҶ’ impossible to topological sort

Use BFS when
вҖў Finding shortest path (unweighted)

Use DFS when

                    count++;          // found a new island

    private void dfs(char[][] grid, int i, int j) {
        // Base case: out of bounds or water
        if (i < 0 || j < 0 || i >= grid.length || j >= grid[0].length || grid[i][j] != '1')

        dfs(grid, i - 1, j); // go up
        dfs(grid, i, j + 1); // go right

    public boolean canFinish(int numCourses, int[][] prerequisites) {
        // Build adjacency list
        List<List<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < numCourses; i++) adj.add(new ArrayList<>());
        
        // Build in-degree array
        int[] inDeg = new int[numCourses];
        for (int[] p : prerequisites) {
            // p = [course, prerequisite] вҶ’ edge: prerequisite вҶ’ course
            adj.get(p[1]).add(p[0]);
            inDeg[p[0]]++;
        }
        
        // Start with all nodes having 0 in-degree
        Queue<Integer> q = new LinkedList<>();
        for (int i = 0; i < numCourses; i++)
            if (inDeg[i] == 0) q.offer(i);
        
        int done = 0;
        while (!q.isEmpty()) {
            int course = q.poll();
            done++;
            for (int neighbor : adj.get(course)) {
                if (--inDeg[neighbor] == 0) q.offer(neighbor);
            }
        }
        return done == numCourses; // if not equal вҶ’ cycle exists
    }
}

Searching (Binary Search)

Where Used

Searching in sorted arrays, finding insertion position, rotated sorted arrays, peak finding, search space reduction (answer lies in a monotonic range), finding boundaries.

Hints to Identify

"sorted array""O(log n)""find position""rotated""peak element""minimize/maximize""first/last occurrence"

Core Logic

Maintain search space [left, right]. Compute mid = left + (right-left)/2 (prevents overflow). Compare mid with target/condition. Eliminate half. Three variants: exact search, left-boundary (first occurrence), right-boundary (last occurrence).

Deep Dive: Binary Search Variants

The Three Templates
// TEMPLATE 1: Exact Match (find target in sorted array)
while (left <= right) {
    int mid = left + (right - left) / 2;
    if (nums[mid] == target) return mid;
    else if (nums[mid] < target) left = mid + 1;
    else right = mid - 1;
}
return -1; // not found

// TEMPLATE 2: Left Boundary (first occurrence of target)
while (left < right) {              // NOTE: < not <=
    int mid = left + (right - left) / 2;
    if (nums[mid] < target) left = mid + 1;
    else right = mid;                 // don't skip mid
}
// After loop: left == right == first occurrence
return nums[left] == target ? left : -1;

// TEMPLATE 3: Right Boundary (last occurrence of target)
while (left < right) {
    int mid = left + (right - left + 1) / 2; // NOTE: +1 to avoid infinite loop
    if (nums[mid] > target) right = mid - 1;
    else left = mid;                          // don't skip mid
}
return nums[left] == target ? left : -1;
Common Pitfalls
Infinite Loop
When left<right and mid=(left+right)/2: if left=mid, it loops. Fix: use +1 in numerator for right-boundary.
Overflow
mid = (left+right)/2 can overflow. Always use: left + (right-left)/2
Wrong Condition
Using <= when you need <, or vice versa. Think: "do I want to include mid in next search?"
Monotonicity
Binary search REQUIRES the condition to be monotonic (sorted, or a yes/no predicate with a single transition point).

        if (left == right) return nums[left]; // base case
        int mid = left + (right - left) / 2;
        int leftMax = divide(nums, left, mid);       // max in left half
        int rightMax = divide(nums, mid + 1, right);  // max in right half
        int crossMax = cross(nums, left, mid, right); // max crossing mid
        return Math.max(Math.max(leftMax, rightMax), crossMax);
    }
    private int cross(int[] nums, int left, int mid, int right) {
        int leftSum = Integer.MIN_VALUE, sum = 0;
        for (int i = mid; i >= left; i--) { sum += nums[i]; leftSum = Math.max(leftSum, sum); }
        int rightSum = Integer.MIN_VALUE; sum = 0;
        for (int i = mid + 1; i <= right; i++) { sum += nums[i]; rightSum = Math.max(rightSum, sum); }
        return leftSum + rightSum;
    }
}
var maxSubArray = function(nums) {
    const divide = (l, r) => {
        if (l === r) return nums[l];
        const mid = Math.floor((l + r) / 2);
        const leftMax = divide(l, mid);
        const rightMax = divide(mid + 1, r);
        let leftSum = -Infinity, sum = 0;
        for (let i = mid; i >= l; i--) { sum += nums[i]; leftSum = Math.max(leftSum, sum); }
        let rightSum = -Infinity; sum = 0;
        for (let i = mid + 1; i <= r; i++) { sum += nums[i]; rightSum = Math.max(rightSum, sum); }
        return Math.max(leftMax, rightMax, leftSum + rightSum);
    };
    return divide(0, nums.length - 1);
};
Hard#50 — Pow(x, n)
Approach

x^n = x^(n/2) × x^(n/2) if n is even, or x × x^(n-1) if n is odd. This reduces O(n) multiplications to O(log n). Handle negative n by computing 1/x^(-n). Use long to avoid overflow when n = Integer.MIN_VALUE.

class Solution {
    public double myPow(double x, int n) {
        long N = n; // use long to handle Integer.MIN_VALUE
        if (N < 0) { x = 1 / x; N = -N; }
        return fastPow(x, N);
    }
    private double fastPow(double x, long n) {
        if (n == 0) return 1.0;
        double half = fastPow(x, n / 2); // divide: solve half
        if (n % 2 == 0) return half * half;         // conquer + combine (even)
        return half * half * x;                      // conquer + combine (odd)
    }
}
var myPow = function(x, n) {
    const fastPow = (x, n) => {
        if (n === 0) return 1;
        const half = fastPow(x, Math.floor(n / 2));
        return n % 2 === 0 ? half * half : half * half * x;
    };
    return n < 0 ? fastPow(1 / x, -n) : fastPow(x, n);
};
Hard#493 — Reverse Pairs
Approach

Modified merge sort. During the merge step, before merging, count how many elements in the right half are less than half of each element in the left half (since condition is i > 2*j). This is D&C because: split array, count inversions in left, count in right, count split inversions during merge.

class Solution {
    public int reversePairs(int[] nums) {
        return mergeSort(nums, 0, nums.length - 1);
    }
    private int mergeSort(int[] nums, int left, int right) {
        if (left >= right) return 0;
        int mid = left + (right - left) / 2;
        int count = mergeSort(nums, left, mid) + mergeSort(nums, mid + 1, right);
        // Count reverse pairs across the two halves
        int j = mid + 1;
        for (int i = left; i <= mid; i++) {
            while (j <= right && (long) nums[i] > 2L * nums[j]) j++;
            count += (j - (mid + 1));
        }
        // Merge step (standard)
        merge(nums, left, mid, right);
        return count;
    }
    private void merge(int[] nums, int left, int mid, int right) {
        int[] temp = new int[right - left + 1];
        int i = left, j = mid + 1, k = 0;
        while (i <= mid && j <= right)
            temp[k++] = nums[i] <= nums[j] ? nums[i++] : nums[j++];
        while (i <= mid) temp[k++] = nums[i++];
        while (j <= right) temp[k++] = nums[j++];
        System.arraycopy(temp, 0, nums, left, temp.length);
    }
}
var reversePairs = function(nums) {
    const mergeSort = (l, r) => {
        if (l >= r) return 0;
        const mid = Math.floor((l + r) / 2);
        let count = mergeSort(l, mid) + mergeSort(mid + 1, r);
        let j = mid + 1;
        for (let i = l; i <= mid; i++) {
            while (j <= r && nums[i] > 2 * nums[j]) j++;
            count += j - (mid + 1);
        }
        // Merge
        const temp = [];
        let i = l; j = mid + 1;
        while (i <= mid && j <= r) temp.push(nums[i] <= nums[j] ? nums[i++] : nums[j++]);
        while (i <= mid) temp.push(nums[i++]);
        while (j <= r) temp.push(nums[j++]);
        for (let k = 0; k < temp.length; k++) nums[l + k] = temp[k];
        return count;
    };
    return mergeSort(0, nums.length - 1);
};
